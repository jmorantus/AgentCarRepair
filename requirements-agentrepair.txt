# Azure AI Foundry AgentCarRepair Web Application Dependencies

# ============================================================================
# CORE DEPENDENCIES (Required)
# ============================================================================

# Flask web framework - Main web application framework
Flask>=3.0.0
    # Flask is the core web framework that provides:
    # - HTTP request/response handling
    # - Template rendering (Jinja2)
    # - Session management
    # - Static file serving
    # - URL routing and view functions

# Azure AI and Identity packages - Core Azure integration
azure-ai-projects>=1.0.0
    # Azure AI Projects SDK provides:
    # - AI Foundry project client
    # - Agent management APIs
    # - Conversation thread handling
    # - Message processing capabilities

    # Azure Identity library provides:
    # - DefaultAzureCredential for automatic auth
    # - ClientSecretCredential for service principal auth
    # - Integration with Azure authentication systems
    # - Token management and refresh

    # - HTTP utilities and data structures
    # - Development server
    # - Debug tools and error handling

    # - HTML template rendering
    # - Template inheritance and macros
    # - Automatic escaping for security

    # - Safe string markup for templates
    # - HTML escaping functions

    # - Flask CLI commands
    # - Application management tools
    # - Custom command creation
